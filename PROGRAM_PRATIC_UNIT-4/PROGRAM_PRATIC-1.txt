#include <stdio.h>

int main() {
    FILE *fp;
    char text[100];

    // Write into file
    fp = fopen("sample.txt", "w");

    printf("Enter text to write in file: ");
    fgets(text, sizeof(text), stdin);

    fprintf(fp, "%s", text);
    fclose(fp);

    // Read from file
    fp = fopen("sample.txt", "r");

    printf("\nContent of file:\n");
    while(fgets(text, sizeof(text), fp) != NULL) {
        printf("%s", text);
    }

    fclose(fp);

    return 0;
}
