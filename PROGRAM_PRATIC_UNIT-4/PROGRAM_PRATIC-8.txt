#include <stdio.h>
#include <string.h>

struct Employee {
    int empcode;
    char empname[50];
    float salary;
};

int main() {
    FILE *fp;
    struct Employee emp;
    int n, i;

    // Step 1: Input number of employees
    printf("Enter number of employees: ");
    scanf("%d", &n);

    // Open binary file in write mode
    fp = fopen("employees.dat", "wb");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }

    // Step 2: Add employee records
    for (i = 0; i < n; i++) {
        printf("\nEnter details for employee %d\n", i + 1);
        printf("Empcode: ");
        scanf("%d", &emp.empcode);
        printf("Empname: ");
        scanf(" %[^\n]", emp.empname); // read string with spaces
        printf("Salary: ");
        scanf("%f", &emp.salary);

        fwrite(&emp, sizeof(struct Employee), 1, fp); // write binary record
    }
    fclose(fp);

    // Step 3: Read the binary file and print records
    fp = fopen("employees.dat", "rb");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }

    printf("\nEmployee Records:\n");
    printf("Empcode\tEmpname\tSalary\n");

    while (fread(&emp, sizeof(struct Employee), 1, fp) == 1) {
        printf("%d\t%s\t%.2f\n", emp.empcode, emp.empname, emp.salary);
    }

    fclose(fp);
    return 0;
}
