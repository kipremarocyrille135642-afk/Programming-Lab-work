#include <stdio.h>

int main() {
    FILE *dataFile, *oddFile, *evenFile;
    int i, num;

    // Step 1: Write numbers 1 to 10 into DATA.txt
    dataFile = fopen("DATA.txt", "w");
    if (dataFile == NULL) {
        printf("Error opening DATA.txt\n");
        return 1;
    }

    for (i = 1; i <= 10; i++) {
        fprintf(dataFile, "%d\n", i);
    }
    fclose(dataFile);

    // Step 2: Open DATA.txt to read
    dataFile = fopen("DATA.txt", "r");
    if (dataFile == NULL) {
        printf("Error opening DATA.txt\n");
        return 1;
    }

    // Open ODD.txt and EVEN.txt for writing
    oddFile = fopen("ODD.txt", "w");
    evenFile = fopen("EVEN.txt", "w");
    if (oddFile == NULL || evenFile == NULL) {
        printf("Error opening ODD.txt or EVEN.txt\n");
        fclose(dataFile);
        return 1;
    }

    // Step 3: Read numbers and store accordingly
    while (fscanf(dataFile, "%d", &num) != EOF) {
        if (num % 2 == 0) {
            fprintf(evenFile, "%d\n", num);
        } else {
            fprintf(oddFile, "%d\n", num);
        }
    }

    // Close all files
    fclose(dataFile);
    fclose(oddFile);
    fclose(evenFile);

    printf("Numbers separated into ODD.txt and EVEN.txt successfully.\n");

    return 0;
}
