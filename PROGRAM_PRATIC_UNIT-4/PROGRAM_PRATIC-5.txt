#include <stdio.h>
#include <string.h>

int main() {
    FILE *fp1, *fp2;
    char str[100];
    int i, len;

    // Take string input
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);

    // Remove newline if present
    len = strlen(str);
    if (str[len - 1] == '\n') {
        str[len - 1] = '\0';
        len--;
    }

    // Write original string into greencity.txt
    fp1 = fopen("greencity.txt", "w");
    if (fp1 == NULL) {
        printf("Error opening greencity.txt\n");
        return 1;
    }
    fputs(str, fp1);
    fclose(fp1);

    // Write reversed string into cleancity.txt
    fp2 = fopen("cleancity.txt", "w");
    if (fp2 == NULL) {
        printf("Error opening cleancity.txt\n");
        return 1;
    }

    for (i = len - 1; i >= 0; i--) {
        fputc(str[i], fp2);
    }

    fclose(fp2);

    printf("Original string written to greencity.txt\n");
    printf("Reversed string written to cleancity.txt\n");

    return 0;
}
