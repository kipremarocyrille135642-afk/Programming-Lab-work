#include <stdio.h>

int main() {
    FILE *fp;
    char text[100];

    // Open file in append mode
    fp = fopen("sample.txt", "a");

    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }

    printf("Enter text to append: ");
    fgets(text, sizeof(text), stdin);

    // Write text at the end of file
    fputs(text, fp);

    printf("Content appended successfully.\n");

    fclose(fp);
    return 0;
}
