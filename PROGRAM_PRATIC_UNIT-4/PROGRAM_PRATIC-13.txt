#include <stdio.h>

int main(int argc, char *argv[]) {
    int i;

    // Print number of arguments
    printf("Number of arguments: %d\n", argc);

    // List all arguments
    printf("Arguments are:\n");
    for (i = 0; i < argc; i++) {
        printf("argv[%d] = %s\n", i, argv[i]);
    }

    return 0;
}
