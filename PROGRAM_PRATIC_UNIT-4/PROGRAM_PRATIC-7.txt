#include <stdio.h>
#include <string.h>

struct Employee {
    int empcode;
    char empname[50];
    float salary;
};

int main() {
    FILE *fp;
    struct Employee emp, highestEmp;
    int n, i;

    // Step 1: Input number of employees
    printf("Enter number of employees: ");
    scanf("%d", &n);

    // Open file in write mode
    fp = fopen("employees.txt", "w");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }

    // Step 2: Add employee records
    for (i = 0; i < n; i++) {
        printf("\nEnter details for employee %d\n", i + 1);
        printf("Empcode: ");
        scanf("%d", &emp.empcode);
        printf("Empname: ");
        scanf(" %[^\n]", emp.empname); // read string with spaces
        printf("Salary: ");
        scanf("%f", &emp.salary);

        fprintf(fp, "%d %s %.2f\n", emp.empcode, emp.empname, emp.salary);
    }
    fclose(fp);

    // Step 3: Read file to find highest salary
    fp = fopen("employees.txt", "r");
    if (fp == NULL) {
        printf("Error opening file.\n");
        return 1;
    }

    highestEmp.salary = -1; // initialize with very small value

    while (fscanf(fp, "%d %s %f", &emp.empcode, emp.empname, &emp.salary) != EOF) {
        if (emp.salary > highestEmp.salary) {
            highestEmp = emp;
        }
    }
    fclose(fp);

    // Step 4: Display employee with highest salary
    printf("\nEmployee with highest salary:\n");
    printf("Empcode: %d\n", highestEmp.empcode);
    printf("Empname: %s\n", highestEmp.empname);
    printf("Salary: %.2f\n", highestEmp.salary);

    return 0;
}
