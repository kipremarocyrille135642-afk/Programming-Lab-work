#include <stdio.h>

struct student {
    int rollno;
    char name[20];
    char department[40];
    int year;
    float score;
};

// Function to print students with score > 5.0
void printTopStudents(struct student s[], int n) {
    int i;

    printf("\nStudents with grade greater than 5.0:\n");
    for(i = 0; i < n; i++) {
        if(s[i].score > 5.0) {
            printf("%s\n", s[i].name);
        }
    }
}

int main() {
    struct student s[5];
    int i;

    // Input 5 student records
    for(i = 0; i < 5; i++) {
        printf("\nEnter details of student %d\n", i + 1);

        printf("Enter Roll No: ");
        scanf("%d", &s[i].rollno);

        printf("Enter Name: ");
        scanf(" %[^\n]", s[i].name);

        printf("Enter Department: ");
        scanf(" %[^\n]", s[i].department);

        printf("Enter Year of Joining: ");
        scanf("%d", &s[i].year);

        printf("Enter Grade: ");
        scanf("%f", &s[i].score);
    }

    printTopStudents(s, 5);

    return 0;
}
