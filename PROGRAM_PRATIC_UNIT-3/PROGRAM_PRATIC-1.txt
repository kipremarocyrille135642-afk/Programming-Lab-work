#include <stdio.h>

// Function to swap values using pointers
void swap(int *a, int *b) {
    int temp;
    temp = *a;
    *a = *b;
    *b = temp;
}

int main() {
    int x, y;

    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);

    printf("Before swap: x = %d, y = %d\n", x, y);

    swap(&x, &y);   // passing addresses

    printf("After swap: x = %d, y = %d\n", x, y);

    return 0;
}
