#include <stdio.h>

// Function to find maximum and minimum using pointers
void findMaxMin(int *a, int *b, int *max, int *min) {
    if (*a > *b) {
        *max = *a;
        *min = *b;
    } else {
        *max = *b;
        *min = *a;
    }
}

int main() {
    int x, y, max, min;

    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);

    findMaxMin(&x, &y, &max, &min);

    printf("Maximum = %d\n", max);
    printf("Minimum = %d\n", min);

    return 0;
}
